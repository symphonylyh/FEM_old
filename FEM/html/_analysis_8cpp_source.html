26 void Analysis::assembleStiffness()
27 {
28  // Initialize global stiffness matrix
29  globalStiffness.resize(2 * mesh.nodeCount(),2 * mesh.nodeCount());
30  // Reserve the space by estimating the maximum number of non-zero elements
31  // in the sparse matrix. This should actually be No. of elements * 2n * 2n
32  // where n is the size of the element type with the most nodes. Here n = 8.
33  int maxNum = mesh.elementCount() * 16 * 16;
34  globalStiffness.reserve(maxNum);
35  globalStiffness.setZero();
36 
37  // Assemble global matrix from local matrix of each element
38  Element* curr;
39  for (int i = 0; i < mesh.elementCount(); i++) {
40  curr = mesh.elementArray()[i];
41  int size = curr->getSize();// element type, for Q4 element, size=4; for Q8, size=8, etc
42  const MatrixXi & nodeList = curr->getNodeList();// the index of nodes belong to this element, e.g., for element8, it will give you a vector contain (10,11,15,14), use this to locate row & column in globalStiffness matrix
43  const MatrixXd & localStiffness = curr->localStiffness();
44  for (int j = 0; j < size; j++){
45  for (int k = 0; k < size; k++){
46  globalStiffness.coeffRef(2 * nodeList(j), 2 * nodeList(k)) += localStiffness(2 * j , 2 * k);
47  globalStiffness.coeffRef(2 * nodeList(j), 2 * nodeList(k) + 1) += localStiffness(2 * j , 2 * k + 1);
48  globalStiffness.coeffRef(2 * nodeList(j) + 1, 2 * nodeList(k)) += localStiffness(2 * j + 1, 2 * k);
49  globalStiffness.coeffRef(2 * nodeList(j) + 1, 2 * nodeList(k) + 1) += localStiffness(2 * j + 1, 2 * k + 1);
50  }
51  }
52  }
53  globalStiffness.makeCompressed();
54 
55  // Sparse matrix operation notes:
56  // m.setZero() to remove all non-zero coefficients
57  // m.rows() to get number of rows
58  // m.cols() to get number of columns
59  // m.coeffRef(i,j) = k to set value to the element already exists;
60  // m.insert(i,j) = k to set value to the element does not already exist;
61  //

99 void Analysis::computeStrainAndStress()
100 {
101  // Idea:
102  // 1. For each element, compute the strain at Gaussian points using nodal
103  // displacments by e = Bu.
104  // 2. Let the nodal strain be unknowns, from equation we can solve for
105  // e(Gaussian) = N e(nodal) --> e(nodal) = N^-1 e(Gaussian)
106  // where N^-1 is the pesudo inverse of the N matrix
107  // 3. Compute stress from strain by sigma = E e
108  // 4. Cumulate strain and stress value at each node, and average at the end
109  // In detail of step 2:
110  // 9 Gaussian points & 8 nodes, we have e(Gaussian) as a 9-by-4 matrix.
111  // N is the stacked shape function [N1 N2 ... N8] at 9 Gaussian points, i.e.
112  // a 9-by-8 matrix. And e(nodal) is the stacked strain vector of 8 nodes as
113  // a 8-by-4 matrix. Since N is not square, pesudo inverse is used to solve
114  // the least square system.
115 
116  nodalStrain.resize(mesh.nodeCount(), 4);
117  nodalStress.resize(mesh.nodeCount(), 4);
118 
119  // Traverse each element
120  Element* curr;
121  int numNodes; // number of nodes belong to the element
122  int numGaussianPt; // number of Gaussian points of the element
123  for (int i = 0; i < mesh.elementCount(); i++) {
124  curr = mesh.elementArray()[i];
125  const VectorXi & nodeList = curr->getNodeList();
126  numNodes = curr->getSize();
127  const std::vector<Vector2d> & GaussianPt = curr->getShape()->gaussianPoint();
128  numGaussianPt = static_cast<int>(GaussianPt.size());
129 
130  // Assemble the nodal displacement vector for an element
131  VectorXd nodeDisp(2 * numNodes);
132  for (int j = 0; j < numNodes; j++) {
133  Vector2d disp = mesh.nodeArray()[nodeList(j)]->getDisp();
134  nodeDisp(2 * j) = disp(0);
135  nodeDisp(2 * j + 1) = disp(1);
136  }
137 
138  MatrixXd strainAtGaussPt(numGaussianPt, 4); // 4 for axisymmetric problem
139  MatrixXd shapeAtGaussPt(numGaussianPt, numNodes); // 9x8 matrix for element Q8
140 
141  // Compute strain at gaussian points from e = Bu
142  for (int g = 0; g < numGaussianPt; g++) {
143  MatrixXd B = curr->BMatrix(GaussianPt[g]);
144  VectorXd e = B * nodeDisp;
145  strainAtGaussPt.row(g) = e.transpose();
146  shapeAtGaussPt.row(g) = curr->getShape()->functionVec(GaussianPt[g]).transpose();
147  }
148 
149  // Solve/extrapolate for nodal strain value via a least square linear system using pesudo inverse
150  MatrixXd pesudo = shapeAtGaussPt.completeOrthogonalDecomposition().pseudoInverse(); // pesudo inverse in "Eigen/QR"
151  MatrixXd strainAtNodes = pesudo * strainAtGaussPt; // 8x4 matrix
152 
153  // Set calculated strain value to 8 nodes
154  const MatrixXd & E = curr->EMatrix();
155  for (int n = 0; n < numNodes; n++) {
156  VectorXd strain = strainAtNodes.row(n);
157  VectorXd stress = E * strain;
158  mesh.nodeArray()[nodeList(n)]->setStrainAndStress(strain, stress);
159  }
160 
161  }
162 
163  // Write nodal strain and stress
164  for (int i = 0; i < mesh.nodeCount(); i++) {
165  nodalStrain.row(i) = mesh.nodeArray()[i]->averageStrain().transpose();
166  nodalStress.row(i) = mesh.nodeArray()[i]->averageStress().transpose();
167  }
168 
169 }
